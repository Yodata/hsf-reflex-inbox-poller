const get = require("lodash.get");
const castArray = require('lodash.castarray');
const debug = require('debug')('red:getRecipients');
 
const isUnacceptedLead = message => {
  return message && message.hasOwnProperty('acceptedByMember') && message.acceptedByMember === false;
};
 
const inboxURL = (recipient) => {
  switch (typeof recipient) {
    case 'string':
      return recipient.replace("/profile/card#me", "/inbox/").toLowerCase();
    case 'object':
      if (recipient == null) return undefined;
      let id = recipient.id || recipient['@id'];
      return (typeof id === 'string') ? id.replace("/profile/card#me", "/inbox/").toLowerCase() : undefined;
    default:
      return undefined;
  }
};
 
function getRecipients(message) {
  if (isUnacceptedLead(message)) {
    debug('Rejecting unaccepted lead');
    return [];
  }
  let recipients = castArray(get(message, "recipient", [])) || [];
  let result = [];
  recipients.forEach(recipient=>{
    let url = inboxURL(recipient);
    url && result.push(url);
  });
  return result;
}
 
module.exports = getRecipients;
